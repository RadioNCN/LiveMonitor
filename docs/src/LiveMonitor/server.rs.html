10pub(crate) async fn GraphServer(db: Arc<DashMap<String,Vec<(f64, f64)>>>,
11                                para: Arc<DashMap<String, pltGraph::Plotpara>>,
12                                data_cap: Arc<Mutex<usize>>) {
13    let server_address = "127.0.0.1:7800";
14    match TcpListener::bind(server_address).await {
15        Ok(listener) => {
16            loop {
17                match listener.accept().await {
18                    Ok((stream, socket)) => {
19                        let db = Arc::clone(&db);
20                        let para = Arc::clone(&para);
21                        let cap = Arc::clone(&data_cap);
22                        tokio::spawn(async move {
23                            let mut reader = BufReader::new(stream);
24                            let mut buffer = String::new();
25                            let mut msg2read = 0i8;
26                            let mut rx_msg = (0f64, 0f64);
27                            let mut stream_id = "temp".to_string();
28                            while let Ok(bytes_read) = reader.read_line(&mut buffer).await {
29                                if bytes_read == 0 {
30                                    println!("Connection was closed");
31                                    break;
32                                }
33                                match msg2read {
34                                    0 => {
35                                        match buffer.trim().parse::<String>() {
36                                            Ok(value) => {
37                                                println!("received Name: {value}");
38                                                stream_id = value;
39                                                if db.contains_key(&stream_id) == false {
40                                                    let _ = db.insert(stream_id.clone(), vec![]);
41                                                    let _ = para.insert(stream_id.clone(), pltGraph::Plotpara::default());
42                                                }
43                                            }
44                                            Err(e) => { println!("could not parse: {}", e) }
45                                        }
46                                        msg2read = 1;
47                                    }
48                                    1 => {
49                                        match buffer.trim().parse::<f64>() {
50                                            Ok(value) => {
51                                                rx_msg.0 = value;
52                                            }
53                                            Err(e) => { println!("could not parse: {}", e) }
54                                        }
55                                        msg2read = 2;
56                                    }
57                                    2 => {
58                                        match buffer.trim().parse::<f64>() {
59                                            Ok(value) => {
60                                                rx_msg.1 = value;
61                                                if let Some(mut entry) = db.get_mut(&stream_id) {
62                                                    entry.push(rx_msg);
63                                                    let cap_lock = cap.lock().await;
64                                                    while entry.len() > *cap_lock {
65                                                        entry.remove(0);
66                                                    }
67                                                    drop(cap_lock);
68                                                }
69                                            }
70                                            Err(e) => { println!("could not parse: {}", e) }
71                                        }
72                                        msg2read = 1;
73                                    }
74                                    _ => {}
75                                }
76                                buffer.clear()
77                            }
78                        });
79                    }
80                    Err(e) => { println!("Can't connect: {}", e) }
81                }
82            }
83        }
84        Err(e) => { println!("Can't bind: {}", e) }
85    }
86}
